pipeline {
    agent any

    tools {
        // Install the Maven version configured as "M3" and add it to the path.
        maven "m3"
    }

    stages {
        stage('Build') {
            steps {
                git 'https://github.com/DavideDeRosa/imolasportiva'

                sh "mvn clean install -DskipTests"
            }

            post {
                success {
                    echo "success"
                    
                    rtUpload serverId: 'artifactory', spec: '''{
                        "files": [
                            {
                                "pattern": "target/*.jar",
                                "target": "imolasportiva/"
                            }
                        ]
                    }'''
                }
                
                failure{
                    echo "failure"
                }
            }
        }
    }
}
